  <script>

    // 1. いただいた鍵でFirebaseに接続

    const firebaseConfig = {

      apiKey: "AIzaSyDiTWsBZSTEAt_xM_SU9yu7QOAAh_c9Cns",

      authDomain: "okurimukae-8a08e.firebaseapp.com",

      projectId: "okurimukae-8a08e",

      storageBucket: "okurimukae-8a08e.firebasestorage.app",

      messagingSenderId: "460205449169",

      appId: "1:460205449169:web:325d073bdbdc84b0134af0"

    };

    firebase.initializeApp(firebaseConfig);

    const db = firebase.firestore();


    // 2. シフト表を作る

    const days = ["月", "火", "水", "木", "金"];

    const times = ["朝", "夕"];

    const shiftTable = document.getElementById("shiftTable");


    days.forEach(day => {

      let tr = document.createElement("tr");

      let tdDay = document.createElement("td");

      tdDay.innerText = day;

      tr.appendChild(tdDay);


      times.forEach(time => {

        let td = document.createElement("td");

        td.id = `shift-${day}-${time}`;

        td.className = "btn-none";

        td.innerText = "未定";

        td.onclick = () => toggleShift(day, time);

        tr.appendChild(td);

      });

      shiftTable.appendChild(tr);

    });


    // 3. シフト変更をデータベースに保存

    async function toggleShift(day, time) {

      const docRef = db.collection("shifts").doc(`${day}-${time}`);

      const currentText = document.getElementById(`shift-${day}-${time}`).innerText;

      let nextUser = currentText === "未定" ? "あや" : (currentText === "あや" ? "けんじさん" : "未定");

      await docRef.set({ user: nextUser });

    }


    // 4. データベースの変更を自動で画面に反映（リアルタイム同期）

    db.collection("shifts").onSnapshot(snapshot => {

      snapshot.docChanges().forEach(change => {

        const docId = change.doc.id; 

        const data = change.doc.data();

        const element = document.getElementById(`shift-${docId}`);

        if(element) {

          element.innerText = data.user;

          element.className = data.user === "あや" ? "btn-aya" : (data.user === "けんじさん" ? "btn-kenji" : "btn-none");

        }

      });

    });


    // 5. チャット機能

    const chatBox = document.getElementById("chatBox");

    const chatInput = document.getElementById("chatInput");


    async function sendMessage() {

      const text = chatInput.value.trim();

      if(!text) return;

      const sender = document.getElementById("userSelect").value;

      await db.collection("chats").add({

        text: text,

        sender: sender,

        timestamp: firebase.firestore.FieldValue.serverTimestamp()

      });

      chatInput.value = "";

    }


    db.collection("chats").orderBy("timestamp").onSnapshot(snapshot => {

      chatBox.innerHTML = "";

      snapshot.forEach(doc => {

        const data = doc.data();

        if(data.text) {

          const wrapper = document.createElement("div");

          wrapper.className = "msg-wrapper";

          const div = document.createElement("div");

          div.className = `msg ${data.sender === "あや" ? "aya" : "kenji"}`;

          div.innerText = `${data.sender}: ${data.text}`;

          wrapper.appendChild(div);

          chatBox.appendChild(wrapper);

        }

      });

      chatBox.scrollTop = chatBox.scrollHeight;

    });
